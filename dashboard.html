""" update_dashboard.py =================== 功能： 1. 找最新的 twse_*.json 2. 把 financial_metrics 嵌入 dashboard.html（替換 SEED_DATA） 3. git commit + push（可開關） 用法： python update_dashboard.py python update_dashboard.py --no-push ← 只更新 HTML，不 push 安全性： - 不刪除任何檔案 - git 用 list 模式，不用 shell=True - 只寫入 dashboard.html，不碰其他檔案 """ import argparse import json import re import subprocess import sys from datetime import datetime from pathlib import Path # ═══════════════════════════════════════════ # CONFIG — 改這裡就好 # ═══════════════════════════════════════════ # twse_fetcher.py 輸出 JSON 的資料夾（你的 JSON 和追蹤個股.txt 在這裡） JSON_FOLDER = Path(r"C:\Users\mis23\OneDrive\桌面\twse_fetcher") # dashboard.html 的位置（GitHub repo 資料夾） DASHBOARD_HTML = Path(r"C:\Users\mis23\OneDrive\桌面\my-stock-dashboard-TWSE-valuation\dashboard.html") # ═══════════════════════════════════════════ # Step 1：找最新 JSON # ═══════════════════════════════════════════ def find_latest_json(folder: Path) -> Path | None: matches = sorted(folder.glob("twse_*.json")) if not matches: print(f"❌ 找不到 twse_*.json 在 {folder}") return None latest = matches[-1] print(f"✅ 使用 JSON：{latest.name}") return latest # ═══════════════════════════════════════════ # Step 2：讀取 financial_metrics # ═══════════════════════════════════════════ def load_metrics(json_path: Path) -> tuple[list, str]: raw = json.loads(json_path.read_text(encoding="utf-8")) metrics = raw.get("financial_metrics", []) # 過濾：pe > 0 且 pb > 0（null 值跳過） valid = [ r for r in metrics if r.get("本益比(倍)") and r.get("股價淨值比(倍)") and r["本益比(倍)"] > 0 and r["股價淨值比(倍)"] > 0 ] date_str = raw.get("meta", {}).get("generated_at", "")[:10] or \ (valid[0].get("資料日期", "") if valid else "") print(f"✅ 有效股票：{len(valid)} 支 / 日期：{date_str}") return valid, date_str # ═══════════════════════════════════════════ # Step 3：產生新的 SEED_DATA JS 陣列 # ═══════════════════════════════════════════ def build_seed_js(metrics: list) -> str: rows = [] for r in metrics: pe = r.get("本益比(倍)") or 0 pb = r.get("股價淨值比(倍)") or 0 dy = r.get("殖利率(%)") or 0 rows.append( f' {{code:"{r["股票代號"]}",name:"{r["公司名稱"]}",' f'pe:{pe},pb:{pb},dy:{dy}}}' ) return "[\n" + ",\n".join(rows) + "\n]" # 替換 let DATA = [ {code:"2330",name:"台積電",pe:29.44,pb:9.33,dy:1.13}, {code:"2317",name:"鴻海",pe:14.86,pb:1.59,dy:3.57}, {code:"2382",name:"廣達",pe:15.81,pb:4.87,dy:5.07}, {code:"3231",name:"緯創",pe:15.2,pb:2.35,dy:4.2}, {code:"2308",name:"台達電",pe:71.95,pb:16.13,dy:0.7}, {code:"3017",name:"奇鋐",pe:45.58,pb:19.61,dy:0.94}, {code:"2356",name:"英業達",pe:17.62,pb:2.01,dy:4.69}, {code:"2357",name:"華碩",pe:9.37,pb:1.56,dy:7.47}, {code:"3711",name:"日月光投控",pe:41.23,pb:4.83,dy:1.72}, {code:"2303",name:"聯電",pe:18.22,pb:2.0,dy:4.31}, {code:"4938",name:"和碩",pe:14.82,pb:1.03,dy:5.03}, {code:"2313",name:"華通",pe:49.73,pb:6.82,dy:1.02}, {code:"2354",name:"鴻準",pe:21.94,pb:0.67,dy:2.95}, {code:"1503",name:"士電",pe:29.74,pb:2.68,dy:2.68}, {code:"1514",name:"亞力",pe:31.97,pb:4.51,dy:2.09}, {code:"2404",name:"漢唐",pe:18.8,pb:9.15,dy:4.42}, {code:"2049",name:"上銀",pe:56.03,pb:2.29,dy:0.83}, {code:"2359",name:"所羅門",pe:90.65,pb:3.31,dy:0.9}, {code:"4585",name:"達明",pe:231.15,pb:4.57,dy:0.33}, {code:"2395",name:"研華",pe:28.5,pb:5.49,dy:3.21}, {code:"2352",name:"佳世達",pe:33.06,pb:1.32,dy:4.2}, {code:"2634",name:"漢翔",pe:60.77,pb:2.5,dy:2.91}, {code:"6753",name:"龍德造船",pe:22.13,pb:3.22,dy:0.97}, {code:"2630",name:"亞航",pe:47.15,pb:2.78,dy:1.91}, {code:"1216",name:"統一",pe:21.07,pb:3.04,dy:4.13}, {code:"6472",name:"保瑞",pe:17.77,pb:3.44,dy:2.4}, {code:"1795",name:"美時",pe:12.05,pb:2.39,dy:1.24} ]; 區塊 pattern = r"let DATA\s*=\s*\[[\s\S]*?\];" replacement = f"let DATA = {seed_js};" new_content, n = re.subn(pattern, replacement, content, count=1) if n == 0: print("❌ 找不到 'let DATA = [...]' 區塊，請確認 dashboard.html 版本正確") return False # 替換 dataDate new_content = re.sub( r'let dataDate\s*=\s*".*?";', f'let dataDate = "2026-04-09";', new_content ) html_path.write_text(new_content, encoding="utf-8") print(f"✅ dashboard.html 已更新（{len(metrics_cache)} 支股票，日期 {date_str}）") return True metrics_cache = [] # ═══════════════════════════════════════════ # Step 5：Git commit + push # ═══════════════════════════════════════════ def git_push(repo_dir: Path, date_str: str) -> bool: commit_msg = f"auto: update dashboard {date_str}" commands = [ ["git", "-C", str(repo_dir), "add", "dashboard.html"], ["git", "-C", str(repo_dir), "commit", "-m", commit_msg], ["git", "-C", str(repo_dir), "push"], ] for cmd in commands: print(f" $ {' '.join(cmd[2:])}") # 只顯示 git 部分 result = subprocess.run(cmd, capture_output=True, text=True, shell=False) out = (result.stdout + result.stderr).strip() if out: print(f" {out}") if result.returncode != 0: if "nothing to commit" in out: print(" ℹ️ Nothing to commit") return True print(f" ❌ git 錯誤") return False return True # ═══════════════════════════════════════════ # MAIN # ═══════════════════════════════════════════ def main(): parser = argparse.ArgumentParser() parser.add_argument("--no-push", action="store_true", help="只更新 HTML，不 push") args = parser.parse_args() print("=" * 50) print(" 台股 Dashboard 更新腳本") print(f" {datetime.now().strftime('%Y-%m-%d %H:%M:%S')}") print("=" * 50) # 1. 找 JSON json_path = find_latest_json(JSON_FOLDER) if not json_path: sys.exit(1) # 2. 讀資料 global metrics_cache metrics, date_str = load_metrics(json_path) metrics_cache = metrics if not metrics: print("❌ financial_metrics 為空，中止") sys.exit(1) # 3. 產生 JS seed_js = build_seed_js(metrics) # 4. 寫入 HTML if not DASHBOARD_HTML.exists(): print(f"❌ 找不到 {DASHBOARD_HTML}") sys.exit(1) if not patch_html(DASHBOARD_HTML, seed_js, date_str): sys.exit(1) # 5. Push if args.no_push: print("\n[--no-push] 跳過 git push") print(f"→ 手動執行：cd {DASHBOARD_HTML.parent} && git add dashboard.html && git commit -m 'update {date_str}' && git push") else: print("\n[Git] commit & push...") ok = git_push(DASHBOARD_HTML.parent, date_str) if ok: print("✅ Push 成功！") print(f"\n🌐 https://mis23ms.github.io/my-stock-dashboard-TWSE-valuation/dashboard.html") else: print("❌ Push 失敗，請手動確認 git 設定") print("\n🎉 完成！") if __name__ == "__main__": main()
